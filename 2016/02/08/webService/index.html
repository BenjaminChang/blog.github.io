@WebService
@BindingType(value = SOAPBinding.SOAP12HTTP_BINDING)
public class UserSynchImpl implements UserSynchI {
private static Logger logger = Logger.getLogger(UserSynchImpl.class);

@Resource(name = "org.apache.cxf.jaxws.context.WebServiceContextImpl")
private WebServiceContext context;

public UserSynchImpl()  {
    System.out.println("userSynchImpl()");
}

@Override
public String doAddOrDelete(String appid,String md5,String status) {

     javax.xml.ws.handler.MessageContext ctx = context
             .getMessageContext();
     HttpServletRequest request = (HttpServletRequest) ctx
             .get(AbstractHTTPDestination.HTTP_REQUEST);
    String ip = request.getRemoteAddr();

    System.out.println("调用doAddOrDelete()时间是："+new SimpleDateFormat("yyyy-MM-dd hh:mm:ss").format(new Date()));
    PropertiesUtil pu = new PropertiesUtil("config.properties") ;
    Gson gson = new Gson() ;
    Map<String,Object> map = new HashMap<String,Object>() ;
    String md5s = Md5Util.md5(pu.getValue("password")) ;
    List<SynchUser> userlist = new ArrayList<SynchUser>();
    if(md5s.equals(md5)){
        System.out.println("有权访问doAddOrDelete()");
        map.put("status", pu.getValue("succCode")) ;
        map.put("mesg", "成功") ;
        UserDao userDao = new UserDao() ;
        userlist = userDao.doAddOrDelete(appid,status);
        map.put("status", pu.getValue("0002")) ;
        map.put("userlist", userlist) ;
    }
    String str = gson.toJson(map) ;
    /*---------------------------------------日志打印----------------------------------*/
    String log = "登录ip为"+ip+"--子系统编号是"+appid+"调用org.jypj.dev.service.UserSynchImpl--doAddOrDelete()--时间是："
            +new SimpleDateFormat("yyyy-MM-dd hh:mm:ss").format(new Date());
    if("1".equals(status)){
        log = log+"--查找新增的用户" ;
    }else{
        log = log+"--查找删除的用户" ;
    }
    logger.info(log);
    String s = "返回的app用户名有这些:" ;
    for(SynchUser u : userlist){
        s = s+u.getApplname()+"," ;
    }
    logger.info(s);
    /*---------------------------------------日志打印----------------------------------*/
    return str;
}
